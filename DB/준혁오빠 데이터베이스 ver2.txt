-- 환자 테이블 생성
CREATE TABLE Patient (
    patient_id INT PRIMARY KEY, -- 환자의 고유 식별자입니다.
    name VARCHAR(255), -- 환자의 이름입니다.
    date_of_birth DATE, -- 환자의 생년월일입니다.(14자리 String)
    gender CHAR(1), -- 환자의 성별입니다 ('M' 또는 'F').
    address VARCHAR(255), -- 환자의 주소입니다.
    phone VARCHAR(20), -- 환자의 전화번호입니다.
    email VARCHAR(320), -- 환자의 이메일 주소입니다.
    blood_type CHAR(2), -- 환자의 혈액형입니다.
    height INT, -- 환자의 키(센티미터)입니다.
    weight INT, -- 환자의 체중(킬로그램)입니다.
    allergies TEXT, -- 환자의 알레르기 정보입니다.
    smoking_status CHAR(1) -- 환자의 흡연 여부입니다 ('Y' 또는 'N').
);

-- 진료과 테이블 생성
CREATE TABLE Department (
    department_id VARCHAR(10) PRIMARY KEY, -- 진료과의 고유 식별자입니다.
    name VARCHAR(100) NOT NULL, -- 진료과의 이름입니다.
    phone VARCHAR(15) NOT NULL -- 진료과의 전화번호입니다.
);

-- 의사 테이블 생성
CREATE TABLE Doctor (
    doc_id CHAR(36) PRIMARY KEY, -- 의사의 고유 식별자입니다.
    doc_name VARCHAR(100), -- 의사의 이름입니다.
    doc_position VARCHAR(10), -- 의사의 직급입니다 (예: 교수, 부교수, 조교수 등).
    department_id VARCHAR(10), -- 의사가 속한 진료과의 식별자입니다.
    email VARCHAR(100) UNIQUE, -- 의사의 이메일 주소입니다.
    profile_image VARCHAR(255), -- 의사의 프로필 이미지 경로입니다.
    doc_shift VARCHAR(100), -- 의사의 근무 교대 정보입니다.
    password VARCHAR(255) DEFAULT '1234', -- 초기 비밀번호는 '1234'로 설정됩니다.
    FOREIGN KEY (department_id) REFERENCES Department(department_id) ON DELETE CASCADE -- 진료과 테이블과의 외래 키 관계입니다.
);

-- 간호사 테이블 생성
CREATE TABLE Nurse (
    nurse_uuid CHAR(36) PRIMARY KEY, -- 간호사의 고유 식별자입니다.
    nurse_pwd VARCHAR(255) NOT NULL, -- 간호사의 비밀번호입니다.
    nurse_role CHAR(1) NOT NULL, -- 간호사의 역할을 나타내는 코드입니다 ('A' - 관리자, 'N' - 일반 간호사).
    nurse_email VARCHAR(320) UNIQUE, -- 간호사의 이메일 주소입니다.
    nurse_name VARCHAR(255), -- 간호사의 이름입니다.
    nurse_shift VARCHAR(100), -- 간호사의 근무 교대 정보입니다.
    department_id VARCHAR(10) NOT NULL, -- 간호사가 속한 진료과의 식별자입니다.
    FOREIGN KEY (department_id) REFERENCES Department(department_id) -- 진료과 테이블과의 외래 키 관계입니다.
);

-- 관리자 테이블 생성
CREATE TABLE Admin (
    admin_id CHAR(36) PRIMARY KEY, -- 관리자의 고유 식별자입니다.
    admin_password VARCHAR(255) NOT NULL, -- 관리자의 비밀번호입니다.
    admin_email VARCHAR(320) UNIQUE -- 관리자의 이메일 주소입니다.
);
