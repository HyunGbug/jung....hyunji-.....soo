-- 진료과 테이블 (Department)
CREATE TABLE Department (
    department_id VARCHAR(10) PRIMARY KEY, -- 진료과의 고유 식별자로, 해당되는 진료과의 약어를 사용합니다.
    name VARCHAR(100), -- 진료과의 이름입니다.
    location VARCHAR(100) -- 진료과의 위치입니다.
);

-- 간호사 테이블 (Nurse)
CREATE TABLE Nurse (
    nurse_id INT PRIMARY KEY, -- 간호사의 고유 식별자입니다.
    name VARCHAR(100), -- 간호사의 이름입니다.
    shift VARCHAR(50), -- 간호사의 근무 시간대입니다.
    department_id VARCHAR(10), -- 간호사가 속한 진료과의 식별자입니다.
    FOREIGN KEY (department_id) REFERENCES Department(department_id) -- 진료과 테이블과의 외래 키 관계
);

-- 환자 테이블 (Patient)
CREATE TABLE Patient (
    patient_id INT PRIMARY KEY, -- 환자의 고유 식별자입니다.
    name VARCHAR(100), -- 환자의 이름입니다.
    date_of_birth DATE, -- 환자의 생년월일입니다.
    gender CHAR(1), -- 환자의 성별입니다.
    address VARCHAR(200), -- 환자의 주소입니다.
    phone VARCHAR(20), -- 환자의 전화번호입니다.
    email VARCHAR(100), -- 환자의 이메일 주소입니다.
    blood_type CHAR(3), -- 환자의 혈액형입니다.
    height INT, -- 환자의 키입니다.
    weight INT, -- 환자의 체중입니다.
    allergies TEXT, -- 환자의 알레르기 정보입니다.
    smoking_status CHAR(1) -- 환자의 흡연 여부입니다.
);

-- 환자 상태 테이블 (PatientConditions)
CREATE TABLE PatientConditions (
    record_id INT PRIMARY KEY, -- 환자 상태 기록의 고유 식별자입니다.
    patient_id INT, -- 환자의 ID를 참조합니다.
    term_id INT, -- 의료 용어의 ID를 참조합니다.
    onset_date DATE, -- 상태가 시작된 날짜입니다.
    status VARCHAR(50), -- 환자의 현재 상태입니다.
    FOREIGN KEY (patient_id) REFERENCES Patient(patient_id) -- 환자 테이블과의 외래 키 관계
);

-- 의사 테이블 (Doctor)
CREATE TABLE Doctor (
    doctor_id INT PRIMARY KEY, -- 의사의 고유 식별자입니다.
    name VARCHAR(100), -- 의사의 이름입니다.
    specialization VARCHAR(100), -- 의사의 전문 분야입니다.
    department_id VARCHAR(10), -- 의사가 속한 진료과의 식별자입니다.
    FOREIGN KEY (department_id) REFERENCES Department(department_id) -- 진료과 테이블과의 외래 키 관계
);

-- 의료 영상 테이블 (MedicalImage)
CREATE TABLE MedicalImage (
    image_id INT PRIMARY KEY, -- 의료 영상의 고유 식별자입니다.
    patient_id INT, -- 영상이 속한 환자의 ID를 참조합니다.
    image_type VARCHAR(50), -- 영상의 종류를 나타냅니다 (예: X-ray, MRI).
    file_path VARCHAR(255), -- 영상 파일의 저장 경로입니다.
    capture_date DATE, -- 영상 촬영 날짜입니다.
    description TEXT, -- 영상에 대한 추가 설명입니다.
    FOREIGN KEY (patient_id) REFERENCES Patient(patient_id) -- 환자 테이블과의 외래 키 관계
);

-- 예약 테이블 (Appointment)
CREATE TABLE Appointment (
    appointment_id INT PRIMARY KEY, -- 예약의 고유 식별자입니다.
    patient_id INT, -- 예약한 환자의 ID입니다.
    doctor_id INT, -- 예약한 의사의 ID입니다.
    appointment_date DATE, -- 예약 날짜입니다.
    appointment_time TIME, -- 예약 시간입니다.
    purpose VARCHAR(255), -- 예약의 목적 또는 진료 내용입니다.
    FOREIGN KEY (patient_id) REFERENCES Patient(patient_id), -- 환자 테이블과의 외래 키 관계
    FOREIGN KEY (doctor_id) REFERENCES Doctor(doctor_id) -- 의사 테이블과의 외래 키 관계
);

-- 관리자 테이블 (Administrator)
CREATE TABLE Administrator (
    admin_id INT PRIMARY KEY, -- 관리자의 고유 식별자입니다.
    name VARCHAR(100), -- 관리자의 이름입니다.
    role VARCHAR(100) -- 관리자의 역할입니다.
);
